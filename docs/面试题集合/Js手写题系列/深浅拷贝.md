```markdown
# 深浅拷贝核心点：浅拷贝只是复制了指向真实数据的指针前后共用同一个内存，而深拷贝相当于是两套数据内存不共用！
```

```js
// 浅拷贝
function shallowClone(o){
  let obj = {}
  for(let i in o){
    obj[i] = o[i]
  }
  return obj
}
// test
let obj = {a:1,b:{c:1,d:{e:2}}}
let obj1 = shallowClone(obj)
obj.b.c.d.e = 6
console.log(obj,obj1) //{a:1,b:{c:1,d:{e:6}}},{a:1,b:{c:1,d:{e:6}}}
```

```js
// 深拷贝
function deepClone(){
  if(typeof obj === 'object' && obj !== null){
   let res = Array.isArray(obj)?[]:{}
 	 for (let key in obj){
    	// 是否有这个属性
    	if(obj.hasOwnProperty(key)){
      		res[key] = deepClone(obj[key])
    	}
  	}
    return res
  }
  return obj
}
// test
let obj = {a:1,b:{c:1,d:{e:2}}}
let obj1 = shallowClone(obj)
obj.b.c.d.e = 6
console.log(obj,obj1) //{a:1,b:{c:1,d:{e:6}}},{a:1,b:{c:1,d:{e:2}}}
```
